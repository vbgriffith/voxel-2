  <script>
    const romInput = document.getElementById("romLoader");

    romInput.addEventListener("change", function () {
      const file = this.files[0];
      const reader = new FileReader();

      reader.onload = function (e) {
        startEmulator(e.target.result);
      };

      reader.readAsDataURL(file);
    });

    function startEmulator(romData) {
      document.getElementById("game").innerHTML = "";

      EJS_player = "#game";
      EJS_core = "snes";
      EJS_gameUrl = romData;
      EJS_pathtodata = "https://cdn.jsdelivr.net/npm/emulatorjs@latest/data/";
      
      // Save system
      EJS_startOnLoaded = true;
      EJS_Settings = {
        saveState: true,
        autoSave: true
      };

      const script = document.createElement("script");
      script.src = "https://cdn.jsdelivr.net/npm/emulatorjs@latest/dist/loader.js";
      document.body.appendChild(script);
    }
  </script>
